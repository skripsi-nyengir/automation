nginx.conf

events {}
http {
  server {
    listen 80;
    root /usr/share/nginx/html;

    location / {
      try_files $uri $uri/ =404;
    }

    # Proxy MQTT WebSocket ke Mosquitto
    location /mqtt {
      proxy_pass http://mosquitto:9001;
      proxy_http_version 1.1;
      proxy_set_header Upgrade $http_upgrade;
      proxy_set_header Connection "upgrade";
      proxy_set_header Host $host;
    }
  }
}

Di app.js — tinggal 1 baris

// Auto-connect — nggak perlu config manual
const brokerUrl = 'ws://' + window.location.host + '/mqtt';
const client = mqtt.connect(brokerUrl);

// Kirim nilai slider ke topic servo
sliderBase.oninput = (e) => {
  client.publish('robot/servo/base', e.target.value);
};

window.location.host = localhost:8080 → otomatis ambil hostname dari URL yang dibuka user.

Nol config. Tinggal docker compose up -d, buka browser, langsung connect.
